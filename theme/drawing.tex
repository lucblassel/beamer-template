% Custom quartet drawing command
\newcommand{\quartet}[5]{%
    \begin{scope}
    \path
    node[anchor=east] (A) at (-1,1) {#1}
    node[anchor=east] (B) at (-1,-1) {#2}
    node[anchor=west] (C) at (1,1) {#3}
    node[anchor=west] (D) at (1,-1) {#4};

    % Add title if wanted
    \ifblank{#5}{}{\node[anchor=north] (cap) at (0,1.5) {(#1#2|#3#4)};}

    \path[shape=coordinate]
    node (u) at (-0.5,0) {}
    node (v) at (0.5,0) {};

    \path[draw,nonzero rule,line cap=butt,line join=round,line width=3pt]
    (A) -- (u) (B) -- (u) (u) -- (v) (v) -- (C) (v) -- (D);
  \end{scope}%
}
